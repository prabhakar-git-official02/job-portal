import multer from "multer";
import path from "path";

const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    if (file.fieldname === "image") {
      cb(null, "uploads/images");
    } else if (file.fieldname === "resume") {
         cb(null, "uploads/resumes");
    } else {
        cb(new Error("Invalid field name"));
    }
  },

  filename: (req, file, cb) => {
    const randomLetters = Math.random().toString(36).substring(2, 8);
    const uniqname =
      Date.now() +
      "-" +
      randomLetters +
      path.extname(file.originalname);
    cb(null, uniqname);
  },
});

const fileFilter = (req, file, cb) => {
  const imageTypes = ["image/jpeg", "image/jpg", "image/png", "image/webp"];
  const resumeTypes = ["application/pdf"];

  if (
    (file.fieldname === "image" && imageTypes.includes(file.mimetype)) ||
    (file.fieldname === "resume" && resumeTypes.includes(file.mimetype))
  ) {
    cb(null, true);
  } else {
    cb(new Error("Only images or PDF resumes are allowed"), false);
  }
};

const upload = multer({
  storage,
  fileFilter,
  limits: {
    fileSize: 5 * 1024 * 1024, // 5mb
  },
});

export default upload;
